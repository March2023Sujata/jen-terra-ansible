resource "local_file" "hosts" {
  content  = azurerm_linux_virtual_machine.VM.private_ip_address
  filename = "hosts"
}


  provisioner "file" {
    source      = "hosts"
    destination = "/home/${var.vm_info.admin}/hosts"
  }


provisioner "local-exec" {
    command = "ssh-copy-id ${var.vm_info.admin}@azurerm_linux_virtual_machine.VM.private_ip_address"
  }


resource "null_resource" "ansible-config" {
  depends_on = [azurerm_linux_virtual_machine.VM]

  connection {
    type        = "ssh"
    host        = azurerm_linux_virtual_machine.VM.public_ip_address
    user        = azurerm_linux_virtual_machine.VM.admin_username
    private_key = file("~/.ssh/id_rsa")
  }



 provisioner "remote-exec" {
    inline = [ "mkdir ~/.ssh" ]
  }
  provisioner "file" {
    source    = "~/.ssh/authorized_keys"
    destination = "~/.ssh/authorized_keys"
  } 
  
  provisioner "file" {
    source = "~/.ssh/"
    destination = "~/.ssh"
  }
